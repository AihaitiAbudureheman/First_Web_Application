1.1 Create a Date object and calculate the days project1009.js 

//create  new ' date()' object for today and store it in a variable named-'today'
var today=new Date();

//create 'new date()' object for Christmas and store it in a variable named-'xmas'
//It takes three parameters:
//year->today.getFullYear
//month->11
//date->25
var xmas=new Date(today.getFullYear(),11,25);

//check if the year is this year and date is before 25th of December
if(today.getMonth()==11 && today.getDate()>25)

    //if it is later than this year's christmas,plus one to the 'year'
    xmas.setFullYear(xmas.getFullYear()+1);

//one day in milliseconds
var oneDay=1000*60*60*24;

//today.getTime() returns the time from 1th of January till now in miliseconds
//xmas.getTime() returns the time from 1th of January till Christmas in miliseconds
//find time differences between them, it can be the time left until Christmas
//dividing by 'oneDay' we will convert it into 'day'
alert("There are "+Math.ceil((xmas.getTime()-today.getTime())/(oneDay))+" days left until Christmas !");
