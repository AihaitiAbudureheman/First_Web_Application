1.1 Get the Date, Month and Year from Date objects project1003.js 

//create a new Date object and store it in a variable named:today
var today= new Date();

//get the Date value of the new Date object by calling
//getDate() function and store it in a variable named:ddd
//The getDate() function returns the day of the month
//for the spesified date according to the local time
//The value returned by getDate() is an integer between
//1 and 31
var ddd=today.getDate();

//get the Month value of the new Date object by calling
//getMonth() function and store it in a variable
//named:mon
//The getMonth() function returns the month of the spesified Date according to the local time as zero based value
//The value returned by getMonth() is an integer between
//0 and 11, 0 corresponds to January, 1 to February, //...11 to December.
//Thats why we plus 1 to the returned value
var mon=today.getMonth()+1;

//get the Year value of the new date object by calling getMonth() function and store it in a variable named:year
//The getYear() function returns the month of the spesified Date according to the local times
//The getYear() function returns absolute numbers between 1000-9999, it is a four digit numbers
var year=today.getFullYear();

//This code will change the 1 into 01,2 into 02,
//...9 into 09
if(ddd<10)
{
    ddd='0'+ddd;
}


//This code will change the 1 into 01,2 into 02,
//...9 into 09
if(mon<10)
{
    mon='0'+mon;
}

//Output dates in different format using alert() function
today=mon+'-'+ddd+'-'+year;//format with '-'
alert(today);

today=mon+'/'+ddd+'/'+year;//format with '/'
alert(today);

today=mon+'.'+ddd+'.'+year;//format with '.'
alert(today);
